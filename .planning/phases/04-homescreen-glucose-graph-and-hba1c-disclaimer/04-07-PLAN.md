      it('returns null for null input', () => {
        expect(glucoseToArcAngle(null)).toBeNull();
      });

      it('returns null for NaN input', () => {
        expect(glucoseToArcAngle(NaN)).toBeNull();
      });

      it('returns -135 for values below minimum (1.0)', () => {
        expect(glucoseToArcAngle(1.0)).toBe(-135);
      });

      it('returns -135 at minimum (2.0 mmol/L)', () => {
        expect(glucoseToArcAngle(2.0)).toBe(-135);
      });

      it('returns 135 at maximum (20.0 mmol/L)', () => {
        expect(glucoseToArcAngle(20.0)).toBe(135);
      });

      it('returns 135 for values above maximum (25.0)', () => {
        expect(glucoseToArcAngle(25.0)).toBe(135);
      });

      it('maps 11.0 mmol/L correctly within arc range', () => {
        const result = glucoseToArcAngle(11.0);
        expect(result).not.toBeNull();
        expect(result!).toBeGreaterThan(-135);
        expect(result!).toBeLessThan(135);
      });

      it('matches formula: -135 + ratio * 270', () => {
        const mmol = 11.0;
        const expected = Math.round(-135 + ((mmol - 2.0) / (20.0 - 2.0)) * 270);
        expect(glucoseToArcAngle(mmol)).toBe(expected);
      });
    });
